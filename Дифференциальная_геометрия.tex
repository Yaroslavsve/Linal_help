    \subsection{Кривые на плоскости}
    Здесь и далее координаты векторов обозначаем верхними индексами, производные обозначаем так: $\dot{a}$.
    \begin{definition}
        Гладкая элементарная регулярная вложенная плоская кривая --- множество точек $\gamma$ на плоскости, задаваемое параметрическими уравнениями
        $$z^1=u(t)$$
        $$z^2=v(t)$$
        $$z(t)=(u(t),v(t))$$
        причём
        \begin{enumerate}
            \item $t\in[a, \ b]$
            \item $z(t)\in C^{\infty}[a,\ b]$ (гладкость)
            \item $\forall t\in[a, \ b] \ \dot{z}(t)=(\dot{u}(t), \ \dot{v}(t))\neq0$ (вектор скорости кривой не обращается в 0 на отрезке $[a, \ b]$ (регулярность))
            \item если $z(t_1)=z(t_2)$, то $t_1=t_2$
        \end{enumerate}
    \end{definition}
    \begin{example}
        $$z^1(t)=t^3$$
        $$z^2(t)=t^2$$
        $$t\in[-1, \ 1]$$
        
    \end{example}
    \begin{exercise}
        Пусть $\gamma$ --- прямой угол, необходимо задать $z=r(t)$, где $r(t)\in C^{\infty}$.\\
    \end{exercise}
    \begin{definition}
        Замена параметра вида: $t(\tau)$, где $\tau\in[\alpha,\ \beta]$, $t(\tau)\in C^{\infty}[\alpha, \ \beta]$, $t'(\tau)\neq 0$, называется допустимой заменой.
    \end{definition}
    \begin{definition}
        Касательной к кривой $\gamma$ в точке $r(t_0)$ называется прямая, проходящая через эту точку в направлении вектора $\dot{r}(t_0)$. $z=r(t_0)+\tau\cdot\dot{r}(t_0)$.
    \end{definition}
    \begin{definition}
        Нормалью к кривой $\gamma$ в точке $r(t_0)$ называется прямая, проходящая через эту точку и ортогональная касательной.
    \end{definition}
    \begin{definition}
        Длиной дуги кривой между точками $t_1$ и $t_2$ называется число $$l=\int\limits_{t_1}^{t_2}|\dot{r}(t)|dt$$
    \end{definition}
    \begin{exercise}
        Доказать, что $l$ не зависит от допустимой параметризации.
    \end{exercise}
    \begin{proof}
        Пусть $t(\tau)$ --- допустимая параметризация, тогда $l=\int\limits_{\tau_1}^{\tau_2}|\dot{r}(\tau)\cdot\frac{d\tau}{dt}|\cdot\frac{dt}{d\tau}d\tau=\int\limits_{\tau_1}^{\tau_2}|\dot{r}(\tau)|d\tau$
    \end{proof}
    Рассмотрим кривую $\gamma \ : \ z=r(t)$, рассмотрим точку $t_0$ на этой кривой и функцию $s(t)=\int\limits_{t_0}^t|\dot{r}(t_1)|dt_1$. Зададим параметризацию $t=t(s)$.
    \begin{definition}
        Такая параметризация называется натуральной параметризацией, а $s$ называется натуральным параметром.
    \end{definition}
    \begin{remark}
        Если кривая $\gamma$ задана натуральной параметризацией, то $$l=|s(t_2)-s(t_1)|$$
    \end{remark}
    \begin{remark}
        $v(s)=1$, обозначение: $\rho'(s):=v(s)$.
    \end{remark}
    Пусть даны две кривые $\gamma$ и $\tilde{\gamma}$, $s_0$ --- их общая точка.
    $$\gamma \ : \ z=\rho(s)$$
    $$\tilde{\gamma} \ : \ z=\tilde{\rho}(s)$$
    $$\rho(s_0)=\tilde{\rho}(s_0)$$
    \begin{definition}
        Кривые $\gamma$ и $\tilde{\gamma}$ имеют в точке $s_0$ касание порядка $k$, если натуральные параметры можно выбрать так, что $\rho(s)-\tilde{\rho}(s)=\overline{\overline{o}}((s-s_0)^k)$.
    \end{definition}
    \begin{subtheorem}
        Кривые $\gamma$ и $\tilde{\gamma}$ имеют в точке $s_0$ касание порядка $k$ тогда и только тогда, когда натуральные параметры можно выбрать так, что $$\rho(s_0)=\tilde{\rho}(s_0), \ \rho'(s_0)=\tilde{\rho}'(s_0), \ \ldots, \ \rho^{(k)}(s_0)=\tilde{\rho}^{(k)}(s_0)$$.
    \end{subtheorem}
    \begin{proof}
        По формуле Тейлора с остаточным членом в форме Пеано $$\rho(s)-\tilde{\rho}(s)=\sum\limits_{j=0}^k\frac{\rho^{(j)}(s_0)-\tilde{\rho}^{(j)}(s_0)}{j!}\cdot(s-s_0)^j+\overline{\overline{o}}((s-s_0)^k)$$
    \end{proof}
    \begin{lemma}
        \hypertarget{d1}{}
        Пусть $\alpha(t)\in\R^n$, $\alpha(t)$ ---бесконечно дифференцируемая вектор-функция и $l=const$, тогда $\alpha\perp\dot{\alpha}$.
    \end{lemma}
    \begin{proof}
        $l^2=(\alpha, \ \alpha)=const$ $\Longrightarrow$ $|\alpha|^2=const$ $\Longrightarrow$ $(|\alpha|^2)'=2(\alpha, \ \dot{\alpha})=0$.
    \end{proof}
    \begin{consequence}
        Если $z=\rho(s)$ --- нормальная параметризация кривой $\gamma$, то $(\ddot{\rho}, \ \dot{\rho})=0$.
    \end{consequence}
    \begin{remark}
        Далее вместо $\dot{\rho}$ пишем $\rho'$, имея ввиду, что $\rho$ --- натуральная параметризация.
    \end{remark}
    \begin{theorem}
        Пусть в некоторой точке $s_0$ кривой $\gamma$ $\rho''(s_0)\neq0$, тогда существует единственная окружность, имеющая касание второго порядка в точке $s_0$, причём центр этой окружности находится на нормали к кривой $\gamma$ в направлении $\rho''(s_0)$, а её радиус $R=\frac{1}{|\rho''(s_0)|}$.
    \end{theorem}
    \begin{proof}
        Рассмотрим окружность $\tilde{\gamma}$ с центром в точке $z_0$ и радиуса $R=\frac{1}{|\rho''(s_0)|}$. Зададим эту окружность так:
        $$\tilde{\gamma} \ : \ z=z_0+\begin{pmatrix}
            a\cos(\frac{s}{a})\\
            a\sin(\frac{s}{a})
        \end{pmatrix}$$
        Тогда $\tilde{\rho}'(s)=\begin{pmatrix}
            -\sin(\frac{s}{a})\\
            \cos(\frac{s}{a})
        \end{pmatrix}$
        $$\tilde{\rho}''(s)=\begin{pmatrix}
            -\frac{1}{a}\cos(\frac{s}{a})\\
            -\frac{1}{a}\sin(\frac{s}{a})
        \end{pmatrix}$$
        Значит, $|\tilde{\rho}''(s)|=\frac{1}{a}$.
    \end{proof}
    \begin{definition}
        Построенная окружность называется соприкасающейся окружностью. Её центр называется центром кривизны, её радиус $R$ --- радиусом кривизны, а величина $k=\frac{1}{R}$ --- кривизной кривой.
    \end{definition}
    \begin{remark}
        Пусть $\ddot{a}(s_0)\neq0$, тогда пара $\{v(s_0), \ n(s_0)=\frac{\ddot{a}(s_0)}{|\ddot{a}(s_0)|}\}$ --- ортонормированный репер с центром в точке $s_0$, $n(s_0)$ --- вектор нормали.
    \end{remark}
    \begin{definition}
        Такой репер называется репером Френе.
    \end{definition}
    \begin{subtheorem}[Плоские формулы Френе]\tab
        \hypertarget{2}{}
        \begin{enumerate}
            \item $v'=\frac{1}{R}\cdot n$
            \item $n'=-\frac{1}{R}\cdot v$
        \end{enumerate}
    \end{subtheorem}
    \begin{proof}\tab
        \begin{enumerate}
            \item $n=\frac{\rho''}{|\rho''|}=\frac{v'}{\frac{1}{R}}$, то есть $v'=\frac{1}{R}\cdot n$.
            \item \hyperlink{d1}{По лемме 1.1} $n'\perp n$, тогда $n' \ \| \ v$ $\Longrightarrow$ $n'=\lambda v$. Так как $(v, \ n)=0$, то\\$(v, \ n')+(v', \ n)=0$. По пункту 1. $v'=\frac{1}{R}\cdot n$, тогда $\frac{1}{R}(n, \ n)+\lambda(v, \ v)=0$.\\Векторы $\vec{n}$ и $\vec{v}$ имеют единичную длину, значит, $\lambda=-\frac{1}{R}$, то есть $n'=-\frac{1}{R}\cdot v$.
        \end{enumerate}
    \end{proof}
    \begin{remark}

            v_1\\
            v_2
        \end{pmatrix}$, тогда рассмотрим\\$\tilde{n}=\begin{pmatrix}
            -v_2\\

        \end{pmatrix}$.\\В таком случае можно не требовать, что вектор скорости нигде не обращается в ноль. При этом $\tilde{k}=\begin{cases}
            k, \ \textup{при } \tilde{n}=n\\
            -k, \ \textup{при } \tilde{n}=-n\\
            0, \ \textup{при } \ddot{a}=0
        \end{cases}$
    \end{remark}
    \begin{subtheorem}\tab
        \begin{enumerate}
            \item Формулы Френе остаются верными при замене $n$ на $\tilde{n}$ и $k$ на $\tilde{k}$.
            \item $\tilde{k}\in C^{\infty}$
        \end{enumerate}
    \end{subtheorem}
    \begin{proof}\tab
        \begin{enumerate}
            \item Так как $\tilde{n}=n$ либо $\tilde{n}=-n$, и знаки $\tilde{n}$ и $\tilde{k}$ совпадают, и $|k|=|\tilde{k}|$, то $kn=\tilde{k}\tilde{n}$.
            \item $|\tilde{n}|=1$, рассмотрим $(v', \ \tilde{n})$. \hyperlink{2}{По первой плоской формуле Френе} $(v', \ \tilde{n})=\tilde{k}\cdot(\tilde{n}, \ \tilde{n})=\tilde{k}=v_1\cdot v_2'-v_2\cdot v_1'$. Так как $v$ --- бесконечно непрерывно дифференцируемая функция, то $\tilde{k}$ тоже.
        \end{enumerate}
    \end{proof}
    \begin{theorem}[о восстановлении кривой по функции кривизны]\tab
        \begin{enumerate}
            \item Пусть $k_0\in C^{\infty}[0, \ l]$, тогда существует кривая $\gamma$ такая, что $k_0$ --- её функция кривизны.
            \item Пусть $\gamma_1$ и $\gamma_2$ --- две кривые такие, что $\tilde{k}_1=\tilde{k}_2$, тогда существует движение $f$ плоскости, переводящее $\gamma_1$ в $\gamma_2$, то есть $f(\gamma_1)=\gamma_2$.
        \end{enumerate}
    \end{theorem}
    \begin{proof}\tab
        \begin{enumerate}
            \item Рассмотрим $\alpha(s)=\int\limits_0^sk_0(s_1)ds_1+\alpha_0$, тогда $\alpha'=k_0$. Рассмотрим $v(s)=\begin{pmatrix}
                \cos(\alpha(s))\\
                \sin(\alpha(s))
            \end{pmatrix}$, тогда $r(s)=\int\limits_0^sv(s_1)ds_1$. Рассмотрим кривую $\gamma \ : \ z=r(s)$. Эта кривая гладкая, так как $v(s)$ --- гладкая функция. Так как $v(s)$ --- бесконечно непрерывно дифференцируемая функция, то $r(s)$ тоже. Из того что $|v(s)|=1\neq 0$, следует, что $\gamma$ регулярна, а так же $r''=v'=\alpha'\cdot\begin{pmatrix}
                -v_2\\
                v_1
            \end{pmatrix}=k_0\cdot\tilde{n}$, то есть $k_0$ --- кривизна кривой $\gamma$.
            \item Соединим начала данных кривых при помощи сдвига. Затем, применим поворот, чтобы их векторы скорости в начальной точке совпали. Так как $\tilde{k}_1=\tilde{k}_2$, то $v_1'=v_2'$, значит, $v_1=v_2+const$. Так как вектора скорости в начальной точке совпадают, то $const=0$. Аналогично $r_1(s)=r_2(s)$. Композиция использованных сдвига и поворота является искомым движением.
        \end{enumerate}
    \end{proof}

    \subsection{Кривые в трёхмерном пространстве}
    \begin{definition}
        Гладкая элементарная регулярная вложенная кривая --- множество точек $\gamma\subset\R^3$, задаваемое параметрическими уравнениями
        $$z^1=z^1(t)$$
        $$z^2=z^2(t)$$
        $$z^3=z^3(t)$$
        $$z(t)=(z^1(t), \ z^2(t), \ z^3(t))$$
        причём
        \begin{enumerate}
            \item $t\in[a, \ b]$
            \item $z(t)\in C^{\infty}[a,\ b]$ (гладкость)
            \item $\forall t\in[a, \ b] \ \dot{z}(t)=(\dot{z}_1(t), \ \dot{z}_2(t), \ \dot{z}_3(t))\neq0$ (вектор скорости кривой не обращается в 0 на отрезке $[a, \ b]$ (регулярность))
            \item если $z(t_1)=z(t_2)$, то $t_1=t_2$
        \end{enumerate}
    \end{definition}
    \begin{definition}
        Касательной к кривой $\gamma$ в точке $r(t_0)$ называется прямая, проходящая через эту точку в направлении вектора $\dot{r}(t_0)$. $z=r(t_0)+\tau\cdot\dot{r}(t_0)$.
    \end{definition}
    \begin{definition}
        Нормальная плоскость к кривой $\gamma$ в точке $r(t_0)$ --- плоскость, проходящая через эту точку и ортогональная касательной. $(z-r(t_0), \ \dot{r}(t_0))=0$.
    \end{definition}
    \begin{remark}
        Длина дуги, натуральная параметризация,  порядок касания, соприкасающиеся окружности, центр и радиус кривизны и кривизна определяются аналогично плоскому случаю.
    \end{remark}
    Пусть $z=r(s)$ и $r(s_0)\neq0$ для некоторого $s_0$, тогда определены $v=r'(s_0)$ и $n=r''(s_0)$
    \begin{definition}
        Вектор $n$ --- вектор главной нормали к кривой. 
    \end{definition}
    Рассмотрим вектор $b=[v, \ n]$.
    \begin{definition}
        Вектор $b$ называется вектором бинормали. Тройка векторов $\{v, \ n, \ b\}$ является ортонормированным репером с центром в точке $s_0$.
    \end{definition}
    \begin{lemma}\hypertarget{3}{}
        Пусть $A(t)$ и $B(t)$ --- это $n\times n$ гладкие матричные функции, заданные на отрезке $[a, \ b]$, причём $\dot{Q}=A\cdot Q$, тогда
        \begin{enumerate}
            \item если $\forall t\in[a, \ b] \ Q(t)\cdot Q^T(t)=E$, то $\forall t\in[a, \ b] \ A^T(t)=-A(t)$.
            \item если $\forall t\in[a, \ b] \ A^T(t)=-A(t)$ --- ортогональная матрица и $Q(t)$ --- ортогональная матрица хотя бы в одной точке, тогда $\forall t\in[a, \ b] \ Q(t)$ --- ортогональная матрица.
        \end{enumerate}
    \end{lemma}
    \begin{proof}
        Рассмотрим $$(Q^T\cdot Q)'=\dot{Q}^T\cdot Q+Q^T\cdot\dot{Q}=(A\cdot Q)^T\cdot Q+Q^T\cdot A\cdot Q=Q^T(A^T+A)Q$$ $Q^T(A^T+A)Q=0 \ \Longleftrightarrow \ A^T+A=0$, то есть $Q^T\cdot Q=const \ \Longleftrightarrow \ A^T+A=0$.
        \begin{enumerate}
            \item Из условия и равносильности выше следует, что $A^T=-A$.
            \item Так как для некоторого $t\in[a, \ b]$ $Q^T(t)\cdot Q(t)=E$, то из равносильности выше это равенство верно для любого $t$.
        \end{enumerate}
    \end{proof}
    \begin{theorem}[\hypertarget{4}{Пространственные формулы Френе}]
        Пусть $\forall s \ r''(s)\neq0$, тогда существует бесконечно непрерывно дифференцируемая функция $\varkappa(s)$ такая, что:
        \begin{enumerate}
            \item $v'=k\cdot n$
            \item $n'=-k\cdot v-\varkappa\cdot b$
            \item $b'=\varkappa\cdot n$
        \end{enumerate}
    \end{theorem}
    \begin{proof}
        Рассмотрим матрицу $Q(s)=\begin{pmatrix}
            \vec{v}\\
            \vec{n}\\
            \vec{b}
        \end{pmatrix}$, тогда $$Q'(s)=\begin{pmatrix}
            \vec{v}'\\
            \vec{n}'\\
            \vec{b}'
        \end{pmatrix}=A\cdot Q$$\hyperlink{3}{По первому пункту леммы 1.2} $A$ --- кососимметрическая матрица.\\Пусть $A=\begin{pmatrix}
            0 & a_{12} & a_{13}\\
            -a_{12} & 0 & a_{23}\\
            -a_{13} & -a_{23} & 0
        \end{pmatrix}$. $n=\frac{r''}{|r''|}=\frac{v'}{k}$, то есть $v'=k\cdot n$. Тогда $a_{12}=k$ и $a_{13}=0$. Таким образом, матрица $A$ имеет вид: $A=\begin{pmatrix}
            0 & k & 0\\
            -k & 0 & a_{23}\\
            0 & -a_{23} & 0
        \end{pmatrix}$. Обозначим $-a_{23}=\varkappa$. Тогда из равенства $Q'=A\cdot Q$ следуют пункты 2 и 3.\\
        Рассмотрим $-(n', \ b)=k\cdot(v, \ b)+\varkappa\cdot(b, \ b)=\varkappa$, значит, $\varkappa(s)$ --- гладкая функция.
    \end{proof}
    \begin{definition}
        Функция $\varkappa(s)$ называется фуннкцией кручения кривой $\gamma$.
    \end{definition}
    \begin{exercise}
        Доказать, что существует вектор $\omega(s)$ такой, что $v'=[\omega, \ v]$,\\$n'=[\omega, \ n]$, $b'=[\omega, \ b]$.
    \end{exercise}
    \begin{proof}
        Рассмотрим вектор $\omega=-\varkappa\cdot v-k\cdot b$ и проверим, что он удовлетворяет нужным условиям.
        \begin{enumerate}
            \item $[\omega, \ v]=[(-\varkappa\cdot v-k\cdot b), \ v]=-\varkappa[v, \ v]-k[b, \ v]=-k[[v, \ n], \ v]$ (из определения вектора $b$), тогда $[\omega, \ v]=-k(v\cdot(n, \ v)-n\cdot(v, \ v))=k\cdot n=v'$.
            \item $[\omega, \ n]=[(-\varkappa\cdot v-k\cdot b), \ n]=-\varkappa[v, \ n]-k[b, \ n]=-\varkappa\cdot b-k[[v, \ n], \ n]$ (из определения вектора $b$), тогда $[\omega, \ n]=-\varkappa\cdot b-k(v\cdot(n, \ n)-n\cdot(v, \ n))=-\varkappa\cdot b-k\cdot v=n'$.
            \item $[\omega, \ b]=[(-\varkappa\cdot v-k\cdot b), \ v]=-\varkappa[v, \ b]-k[b, \ b]=-\varkappa[v, [v, \ n]]$ (из определения вектора $b$), тогда $[\omega, \ b]=-\varkappa(v\cdot(v, \ n)-n\cdot(v, \ v))=\varkappa\cdot n=b'$.
        \end{enumerate}
    \end{proof}
    \begin{definition}
        Кривая $\gamma$ бирегулярна, если $\forall s \ r''(s)\neq0$.
    \end{definition}
    \begin{subtheorem}
        Пусть кривая $\gamma$ бирегулярна, тогда она плоская тогда и только тогда, когда $\varkappa\equiv0$.
    \end{subtheorem}
    \begin{proof}

        $\underline{\Longleftarrow}$ Так как $\varkappa\equiv0$, то $b'\equiv0$ $\Longrightarrow$ $b=b_0=const$. По определению $(b, \ v)=0$, тогда $0=(b_0, \ r')=(b_0, \ r)'$ $\Longrightarrow$ $(b_0, \ r)=const=c_0$.  Последнее уравнение является уравнением плоскости, значит, кривая $\gamma$ плоская.
    \end{proof}

    \section{Дифференциальная геометрия поверхностей}
    Будем рассматривать поверхности размерности $n$ в евклидовом пространстве размерности $N$.
    \begin{definition}
        Гладкая регулярная вложенная элементарная $n$-мерная поверхность в евклидовом пространстве размерности $N$ --- это множество точек, задаваемое параметрическим уравнением $z=r(x)$, причём должны быть выполнены следующие условия:
        \begin{enumerate}
            \item $x=(x^1, \ldots, \ x^n)\in D$ (некоторая открытая область)
            \item $r(x)\in C^{\infty}(D)$
            \item $r_1=\frac{\partial r}{\partial x^1}, \ \ldots, \ r_n=\frac{\partial r}{\partial x^n}$ линейно независимы в области $D$
            \item $r(x)=r(x^*) \ \Longleftrightarrow \ x=x^*$
        \end{enumerate}
        Параметры $x^1$, $\ldots$, $x^n$ называются координатами на этой поверхности.
    \end{definition}
    \begin{example}[не регулярной поверхности]
        Рассмотрим $\begin{cases}
            z^1=(x^1)^3\\
            z^2=(x^1)^2\\
            z^3=x^2
        \end{cases}$\\
        Тогда $r_1=\begin{pmatrix}
            3(x^1)^2\\
            2x^1\\
            0
        \end{pmatrix}$, $r_2=\begin{pmatrix}
            0\\
            0\\
            1
        \end{pmatrix}$, значит, в нуле векторы будут линейно зависимы.
    \end{example}
    \begin{remark}
        Пусть $y\in\tilde{D}\subset\R^n$, $x=x(y)$ --- биекция между $\tilde{D}$ и $D$, $x(y)$ и $y(x)$ --- гладкие функции, $det\left(\left(\frac{\partial x^i}{\partial y^j}\right)\right)\neq0$ , тогда $z=\rho(y)=r(x(y))$.
        $$\frac{\partial\rho}{\partial y^j}=\sum\limits_{k=1}^n\frac{\partial r}{\partial x^k}\cdot\frac{\partial x^k}{\partial y^j}$$
        Отсюда следует, что новая поверхность также регулярна, а значит, её можно определить как класс эквивалентности всевозможных таких параметризаций. Такое определение не требует 4 пункта исходного определения.
    \end{remark}
    \begin{definition}
        Гладкая кривая $\gamma$ на поверхности $M$ --- это гладкое отображение $\gamma:[a, \ b]\to M$, то есть соответствие, сопоставляющее каждой точке $t\in[a, \ b]$ точку $\gamma(t)\in M$, причём если $x=(x^1, \ \ldots, \ x^n)$ --- точка на $M$, то соответствующие функции $x(t)=(x^1(t), \ \ldots, \ x^n(t))$ являются гладкими.
    \end{definition}
    \begin{definition}
        Касательный вектор к $M$ в точке $P$ --- это вектор скорости некоторой кривой, лежащей на $M$ и проходящей через точку $P$.
    \end{definition}
    \begin{subtheorem}
        \hypertarget{5}{}
        Множество всех касательных векторов к $M$ в точке $P$ образует векторное пространство, причём базисом этого пространства являются векторы $r_1(P)$, $\ldots$, $r_n(P)$, где $r_i=\frac{\partial r}{\partial x^i}$.
    \end{subtheorem}
    \begin{proof}
        Из определения следует, что векторы $r_1(P)$, $\ldots$, $r_n(P)$ линейно независимы, а значит, их линейная оболочка образует векторное пространство.\\
        Пусть $\gamma$ --- кривая на $M$, $x=x(t)$, $x_0=x(t_0)=P$, $z=r(x(t))=\rho(t)$.\\Тогда $\dot{\rho}(P)=\sum\limits_{j=1}^n\frac{\partial r}{\partial x^j}(P)\cdot \dot{x}^j(t_0)$, то есть этот вектор является линейной комбинацией векторов $r_1(P)$, $\ldots$, $r_n(P)$.\\
        Пусть $X=\sum\limits_{j=1}^nX^j\cdot r_j(P)$, $x^j=x_0^j+tX^j$.\\Рассмоторим $\dot{\gamma}=\sum\limits_{j=1}^n\dot{x}^j\cdot r_j(P)=\sum\limits_{j=1}^nX^j\cdot r_j(P)=X$.\\Значит, по определению, $r_1$, $\ldots$, $r_n$ --- базис.
    \end{proof}
    \begin{definition}
        Это множество называется касательным пространством к $M$ в точке $P$ и обозначается $T_PM$.
    \end{definition}
    \begin{remark}
        Пусть $x=x(y)$, $z=\rho(y)=r(x(y))$, тогда $\rho_j=\frac{\partial\rho}{\partial y^j}=\sum\limits_{k=1}^n\frac{\partial x^k}{\partial y^j}\cdot r_k$. Отсюда следует, что касательное пространство инвариантно относительно замены параметра, изменится только базис.
    \end{remark}
    \begin{definition}
        Первая фундаментальная форма поверхности --- это соответствие, сопоставляющее каждой точке $P$ этой поверхности симметричную положительно определённую билинейную форму в $T_PM$ --- ограничение скалярного произведения из $\R^N$.
    \end{definition}
    \begin{remark}\tab
        \begin{enumerate}
            \item Пусть $X$, $Y\in T_PM$, $x=\sum\limits_{j=1}^nx^jr_j$, $x=(x^1, \ \ldots, \ x^n)$, тогда $(X, \ Y)=\sum\limits_{i,j=1}^{n}g_{ij}x^iy^j$
            \item $|X|=\sqrt{(X, \ X)}$, $\cos(\phi_{xy})=\frac{(X, \ Y)}{|X|\cdot|Y|}$
            \item Пусть $\gamma$ --- кривая на поверхности $M$, $\gamma :x^i=x^i(t)$, $t\in[a, \ b]$. $\dot{\gamma}=(\dot{x}^1, \ \ldots, \ \dot{x}^n)$, тогда $l=\int\limits_a^b|\dot{\gamma}|dt=\int\limits_a^b\sqrt{\sum\limits_{i,j=1}^{n}g_{ij}(x(t))\cdot\dot{x}^i\dot{x}^j}dt$
            \item Пусть $\gamma$ и $\tilde{\gamma}$ --- две кривые на поверхности $M$ и пусть они пересекаются в точке $P$. $\gamma: x^i=x^i(t)$, $\tilde{\gamma}:\tilde{x}^i=\tilde{x}^i(t)$. Так как кривые пересекаются, то $x^i(t_0)=\tilde{x}^i(\tau_0)$ для некотырых $t_0$ и $\tau_0$. Тогда $\cos(\phi)=\frac{(\dot{\gamma}, \ \dot{\tilde{\gamma}})}{|\dot{\gamma}|\cdot|\dot{\tilde{\gamma}}|}$.
        \end{enumerate}
    \end{remark}
    \begin{definition}
        Ковариантное ускорение кривой $\gamma$ --- это ортогональная проекция вектора ускорения на касательную плоскость $T_{\gamma}M$. Обозначается $\nabla_{\dot{\gamma}}\dot{\gamma}$.
    \end{definition}
    \begin{definition}
        Кривая называется геодезической, если $\nabla_{\dot{\gamma}}\dot{\gamma}\equiv0$.
    \end{definition}
    \subsection{Ковариантное дифференцирование векторных полей}
    \begin{definition}
        Пусть $M$ --- $n$-мерная поверхность. $f:M\to\R$ --- гладкая функция, если для любой системы координат $x^1, \ \ldots, \ x^n$ $f(x^1, \ \ldots, \ x^n)\in C^{\infty}$.
    \end{definition}
    \begin{definition}
        Пусть $P\in M$ --- точка на поверхности $M$, $X\in T_PM$ --- касательный вектор. Рассмотрим кривую $\gamma:[a, \ b]\to M$ такую, что $\gamma(t_0)=P$ и $\dot{\gamma}(t_0)=X$ для некоторого $t_0\in[a, \ b]$. Производной функции $f$ вдоль вектора $X$ называется $\partial_Xf=\frac{\partial}{\partial t}f(\gamma(t))$.
    \end{definition}
    \begin{subtheorem}
        $\partial_Xf$ не зависит от $\gamma$.
    \end{subtheorem}
    \begin{proof}
        Пусть $z=r(x)$, $\gamma: x=x(t)$, $f=f(x)$, $r_1$, ..., $r_n$ --- базис касательного пространства, тогда $X=\sum\limits_{i=1}^nX^ir_i(P)$, $X^i=\dot{x}^i$
        $$\partial_Xf=\frac{\partial}{\partial t}\vert_{t=t_0}f=\sum\limits_{i=1}^{n}X^i\frac{\partial f}{\partial x^i}$$
        Легко видеть, что полученное выражение не зависит от кривой $\gamma$.
    \end{proof}
    \begin{definition}
        Гладкое векторное поле на поверхности $M$ --- это соответствие, сопоставляющее каждой точке $P\in M$ вектор $X(P)\in T_PM$.
    \end{definition}
    \begin{definition}
        Ковариантная производная поля $X$ в точке $P$ вектора $Y$ --- это $\nabla_YX=\Pi\left(\frac{\partial}{\partial t}\vert_{t=t_0}X(\gamma(t))\right)$.
    \end{definition}
    \begin{subtheorem}
        Ковариантная производная не зависит от $\gamma$.
    \end{subtheorem}
    \begin{proof}
        $M:z=r(x)$ --- заданная поверхность, $X=\sum\limits_{i=1}^nX^i(x)\cdot r_i(x)$, $Y=\sum\limits_{j=1}^nY^j(P)\cdot r_j(P)$, $\gamma:x=x(t)$, $Y^j=\dot{x}^j(t_0)$, где $x(t_0)=P$.\\
        Тогда $$\nabla_YX=\Pi\left(\frac{d}{dt}\vert_{t=t_0}\sum\limits_{i=1}^nX^i(x(t))r_i(x(t))\right)=\Pi\left(\sum\limits_{i=1}^n\left(\frac{\partial X^i}{\partial t}r_i+X^i\cdot\frac{\partial r^i}{\partial t}\right)\right)=$$
        $$=\Pi\left(\sum\limits_{i=1}^n\left(\frac{\partial X^i}{\partial x^j}\cdot\dot{x}^jr_i+X^i\cdot\frac{\partial r_i}{\partial x^j}\dot{x}^j\right)\right)=Y^j\cdot\Pi\left(\sum\limits_{i=1}^n\left(\frac{\partial X^i}{\partial x^j}\cdot r_i+X^i\cdot\frac{\partial r_i}{\partial x^j}\right)\right)$$
        Легко видеть, что последнее выражение не зависит от кривой $\gamma$, значит, ковариантная производная не зависит от кривой $\gamma$.
    \end{proof}
    \begin{remark}
        Пусть $r_{ij}=\frac{\partial^2 r}{\partial x^i\partial x^j}=\frac{\partial r_i}{\partial x^j}$, тогда, пользуясь тем, что векторы $r_i$ образуют базис касательного пространства \hyperlink{5}{по утверждению 2.1 и соответствующему определению}, получим $$\nabla_YX=Y^j\sum\limits_{i=1}^n\left(\frac{\partial X^i}{\partial x^j}r_i+\Pi\left(X^i\frac{\partial r_i}{\partial x^j}\right)\right)=Y^j\sum\limits_{k=1}^n\left(\frac{\partial X^k}{\partial x^j}r_k+X^k\Pi(r_{jk})\right)$$
        Сделаем замену $\Pi(r_{jk})=\sum\limits_{i=1}^n\Gamma_{jk}^i(x)r_i$. Тогда формула ковариантной прямой будет иметь вид:
        $$Y^j\sum\limits_{i=1}^n\left(\frac{\partial X^i}{\partial x^j}+\sum\limits_{k=1}^n\left(\Gamma_{jk}^i(x)\cdot X^k\right)\right)r_i$$
    \end{remark}
    \begin{definition}
        Коэффициенты $\Gamma_{jk}^i$ называются символами Кристоффеля второго рода, а коэффициенты $\Gamma_{ij, \ m}=(r_{ij}, \ r_m)$ --- символами Кристоффеля первого рода.
    \end{definition}
    \begin{subtheorem}\hypertarget{10}{}
        $\Gamma_{ij, \ m}=\frac{1}{2}\left(\frac{\partial g_{im}}{\partial x^j}+\frac{\partial g_{jm}}{\partial x^i}-\frac{\partial g_{ij}}{\partial x^m}\right)$, где $g_{ij}=((r_i, \ r_j))$ --- матрица первой фундаментальной формы.\\
        $\Gamma_{ij}^k=\frac{1}{2}g^{km}\left(\frac{\partial g_{im}}{\partial x^j}+\frac{\partial g_{jm}}{\partial x^i}-\frac{\partial g_{ij}}{\partial x^m}\right)$, где $g^{km}=g_{ij}^{-1}$.
    \end{subtheorem}
    \begin{proof}
        $$\frac{\partial g_{ij}}{\partial x^m}=(r_{im}, \ r_{j})+(r_{jm}, \ r_i)=\Gamma_{im, \ j}+\Gamma_{jm, \ i}$$
        Переставим переменные по циклу $(i, \ j, \ m)$
        $$\frac{\partial g_{jm}}{\partial x^i}=(r_{ji}, \ r_{m})+(r_{mi}, \ r_j)=\Gamma_{ji, \ m}+\Gamma_{mi, \ j}$$
        $$\frac{\partial g_{mi}}{\partial x^j}=(r_{mj}, \ r_{i})+(r_{ij}, \ r_m)=\Gamma_{mj, \ i}+\Gamma_{ij, \ m}$$
        Отсюда получим
        $$\frac{\partial g_{im}}{\partial x^j}+\frac{\partial g_{jm}}{\partial x^i}-\frac{\partial g_{ij}}{\partial x^m}=\Gamma_{jm, \ i}+\Gamma_{ij, \ m}+\Gamma_{ij, \ m}+\Gamma_{im, \ j}-\Gamma_{im, \ j}-\Gamma_{jm, \ i}=2\Gamma_{ij, \ m}$$
        Введём обозначения: $\hat{\Gamma}_{ij}=\begin{pmatrix}
            \Gamma_{ij}^1\\
            \vdots\\
            \Gamma_{ij}^n
        \end{pmatrix}$, $\hat{G}_{ij}=\begin{pmatrix}
            \Gamma_{ij, \ 1}\\
            \vdots\\
            \Gamma_{ij, \ n}
        \end{pmatrix}$
        $$\Gamma_{ij, \ m}\overset{(1)}{=}(r_{ij}, \ r_m)\overset{(2)}{=}(\Pi(r_{ij}), \ r_m)\overset{(3)}{=}\left(\sum\limits_{k=1}^n\Gamma_{ij}^kr_k, \ r_m\right)=\sum\limits_{k=1}^ng_{km}\Gamma_{ij}^k$$
        (1) : по определению\\
        (2) : так как векторы $r_m$ являются векторами касательного пространства\\
        (3) : по определению
        $$\hat{G}_{ij}=g_{km}\hat{\Gamma}_{ij} \ \Longrightarrow \ \Gamma_{ij}^k=\frac{1}{2}g^{km}\left(\frac{\partial g_{im}}{\partial x^j}+\frac{\partial g_{jm}}{\partial x^i}-\frac{\partial g_{ij}}{\partial x^m}\right)$$
    \end{proof}
    \subsubsection{Параллельный перенос касательных векторов}
    \begin{definition}
        Гладким семейством векторов, касательных к $M$ в точках кривой $\gamma$, называется соответствие, сопоставляющее каждому $t\in[a, \ b]$ вектор $X(t)\in T_{\gamma(t)}M$ и гладко зависящее от $t$.
    \end{definition}
    \begin{definition}
        Ковариантная производная гладкого семейства $X(t)$ в точке $t_0$ вдоль кривой $\gamma$ --- это $\nabla_{\dot{\gamma}}X=\Pi(\dot{X}(t_0))$.
    \end{definition}
    Пусть $x=(x^1, \ \ldots, x^n)$, $\gamma: x=x(t)$, $X=\sum\limits_{i=1}^nX^i(t)r_i(x(t))$, тогда $$\nabla_{\dot{\gamma}(t)}X=\Pi\left(\sum\limits_{i=1}^n(\dot{X}^ir_i)+\sum\limits_{k=1}^nX^k\frac{\partial r_k}{\partial x^j}\dot{x}^j\right)=$$

    Последняя формула является формулой ковариантной производной.

    \end{definition}

    \end{remark}

            \item $X(a)=X_0$
        \end{enumerate}
    \end{subtheorem}
    \begin{proof}

            X^i(a)=X_0^i\\
            \dot{X}^i+\sum\limits_{k=1}^n\Gamma_{jk}^i\dot{x}^jX^k=0
        \end{cases}$$
        Данная система является системой линейных однородных дифференциальных уравнений с начальными условиями, значит, по теореме Коши она имеет единственное решение.
    \end{proof}

    \end{definition}

        Поверхности $M$ и $N$ касаются вдоль кривой $\gamma$, если $\gamma$ лежит на $M$ и на $N$ и $\forall t \ T_{\gamma(t)}M=T_{\gamma(t)}N$
    \end{definition}

        \hypertarget{6}{}

            \item Параллельный перенос линейно зависит от $X_0$.
            \item Параллельный перенос сохранят скалярное произведение.
            \item Если $M$ и $N$ --- касаются вдоль кривой $\gamma$, то $X(t)$ являестя результатом параллельного переноса вдоль кривой $\gamma$ на $M$ тогда и только тогда, когда оно является результатом параллельного переноса вдоль кривой $\gamma$ на $N$.
        \end{enumerate}
    \end{propertes}
    \begin{proof}\tab
        \begin{enumerate}
            \item Очевидно следует из $\dot{X}^i+\sum\limits_{k=1}^n(\Gamma_{jk}^i\dot{x}^jX^k)=0$
            \item Рассмотрим $X(t)$ и $Y(t)$ вдоль кривой $\gamma$, тогда $((X, \ Y))'_t=(\dot{X}, \ Y)+(X, \ \dot{Y})=$ $=(\Pi(\dot{X}), \ Y)+(X, \ \Pi(\dot{Y}))=(\nabla_{\dot{\gamma}}X, \ Y)+(\nabla_{\dot{\gamma}}Y, \ X)=0$
            \item $\forall X\in T_{\gamma(t)}M$ $X\in T_{\gamma(t)}N$, $\nabla_{\dot{\gamma}}X=\Pi(\dot{X})$
        \end{enumerate}
    \end{proof}

            \item Через каждую точку в направлении каждого касательного вектора проходит одна геодезическая кривая.

            \item При параллельном переносе вектора $X(t)$ вдоль геодезической кривой сохраняется угол между этой кривой и вектором $X$.
            \item Если поверхности $M$ и $Q$ касаются вдоль кривой $\gamma$, то $\gamma$ геодезическая на $M$ тогда и только тогда, когда $\gamma$ геодезическая на $Q$.
        \end{enumerate}

        \begin{enumerate}
            \item Следует из теоремы Коши.

Значит, $|\dot{\gamma}|^2=const$, то есть $|\dot{\gamma}|=const$.

            \item \hyperlink{6}{Из свойства 3 параллельного переноса} и из свойства 2 геодезической кривой следует данное утверждение.
        \end{enumerate}
    \end{proof}

    \begin{definition}
        Пусть $M\subset\R^n$ и $Q\subset\R^k$ --- поверхности в $\R^N$. Пусть $M:z=r(x)$, $x=(x^1, \ \ldots, \ x^n)$, $Q:w=\rho(y)$, $y=(y^1, \ \ldots, \ y^k)$. Рассмотрим отображение $f:M\to Q$ такое, что $y^i=f^i(x^1, \ \ldots, \ x^n)$ $\forall i=\overline{1,k}$. Тогда отображение $f$ называется гладким, если $\forall i$, $f^i\in C^{\infty}$.
    \end{definition}
    \begin{subtheorem}
        Пусть $X=\dot{\gamma}(t_0)$, $Y=\frac{d}{dt}|_{t=t_0}f(\gamma)$. Тогда $Y$ не зависит от выбора кривой $\gamma$. Более того, соответствие $X\longrightarrow Y$ является линейным отображением из $T_PM$ в $T_{f(P)}Q$.
    \end{subtheorem}
    \begin{proof}
        Пусть $\gamma:x^i=x^i(t)$, $X^i=\dot{x}^i(t_0)$, $x=(x^1, \ \ldots, \ x^n)$, $f(\gamma):y^i=f^i(x(t))$, $Y^i=\frac{d}{dt}|_{t=t_0}f^i(x(t))$, $y=(y^1, \ \ldots, \ y^k)$. Тогда $Y^i=\frac{\partial f^i}{\partial x^j}(P)\cdot\dot{x}^j(t_0)=\frac{\partial f^i}{\partial x^j}(P)\cdot X^j$
    \end{proof}
    \begin{definition}
        Построенное отображение называется дифференциалом $f$ в точке $P$ и обозначается $Y=d_Pf(X)$.
    \end{definition}
    \begin{definition}
        Гладкое отображение $f:M\to Q$ называется изометрией, если:
        \begin{enumerate}
            \item $f$ --- биекция, $f^{-1}$ --- гладкое отображение, то есть $f$ --- диффеоморфизм.
            \item $\forall P\in M$, $X$, $Y\in T_PM$ $(X, \ Y)=(d_Pf(X), \ d_Pf(Y))$.
        \end{enumerate}
    \end{definition}
    \begin{definition}
        Поверхности $M$ и $Q$ называются изометричными, если существует изометрия $f:M\to Q$.
    \end{definition}
    \begin{exercise}
        Доказать следующие свойства изометрии:
        \begin{enumerate}
            \item Изометрия сохраняет длины кривых и углы между кривыми.
            \item Параллельные семейства векторов переходят в параллельные.
            \item Геодезические кривые переходят в геодезические.
        \end{enumerate}
    \end{exercise}
    \begin{proof}\tab
        \begin{enumerate}
            \item $|X|^2=(X, \ X)=(d_Pf(X), \ d_Pf(X))=|d_PX|^2$, $\cos(\phi_1)=\frac{(d_Pf(X), \ d_Pf(Y))}{|d_Pf(X)|\cdot|d_Pf(Y)|}=\frac{(X, \ Y)}{|X|\cdot|Y|}=\cos(\phi_0)$. Значит, длина вектора не меняется при изометрии, а значит, $|\dot{\gamma}|$ не меняется, следовательно по определению длина кривой тоже не меняется. Так как углы между векторами сохраняются, то и углы между кривыми сохраняются.
            \item Два семейства векторов $\{X_t\}$ и $\{Y_t\}$ называются параллельными, если $\forall t$ $X_t\upuparrows Y_t$, то есть существует гладкая функция $\lambda(t)$ такая, что $X_t=\lambda(t)Y_t$. Рассмотрим $d_Pf(X_t)=d_Pf(\lambda(t)Y_t)=\lambda(t)d_Pf(Y_t)$, тогда по второму пункту определения получим данное утверждение.
            \item Пусть кривая $\gamma$ геодезическая, то есть $\Pi(\ddot{\gamma})=0$. Рассмотрим $X\in T_PM$ $\frac{d}{dt}(\dot{\gamma}, \ X)=(\Pi(\ddot{\gamma}), \ X)+(\dot{\gamma}, \ \Pi(\dot{X}))=0$, $(\dot{\gamma}, \ X)=(d_Pf(\dot{\gamma}), \ d_Pf(X))=const$.
        \end{enumerate}
    \end{proof}

    \begin{proof}

Для этого задаим параметризацию $r(\alpha, \ h)=\begin{pmatrix}
            R\cos\alpha\\
            R\sin\alpha\\
            h
        \end{pmatrix}$ (стандартные цилиндрические координаты с центром в точке $A$). Зададим плоскость, начало координат на которой помещено в точку $A$, ось $Oy$ будет сонаправлена с вектором $\overline{AB}$, а ось $Ox$ выберем так, чтобы выполнялась положительная ориентация. Тогда развёрткой будет являться отображение, которое каждой точке с координатами $(\alpha, \ h)$, будет сопоставлять точку на плоскости с координатами $(\alpha R, \ h)$, где $R$ --- радиус цилиндра. Данное отображение биективно, непрерывно, обратное отображение тоже непрерывно, то есть остаётся проверить сохранение скалярного произведения для дифференциала этого отображения. Для этого вычислим первую фундаментпльную форму. На цилиндре
        $$\frac{\partial r}{\partial\alpha}=\begin{pmatrix}
            -R\sin\alpha\\
            R\cos\alpha\\
            0
        \end{pmatrix}, \ \frac{\partial r}{\partial h}=\begin{pmatrix}
            0\\
            0\\
            1
        \end{pmatrix}$$
        $$g=\begin{pmatrix}
            R^2 & 0\\
            0 & 1
        \end{pmatrix}$$
        На плоскости $g=E=\begin{pmatrix}
            1 & 0\\
            0 & 1
        \end{pmatrix}$.
        Матрица дифференциала (матрица якоби) будет иметь вид $$\begin{pmatrix}
            \frac{\partial x}{\partial\alpha} & \frac{\partial x}{\partial h}\\
            \frac{\partial y}{\partial\alpha} & \frac{\partial y}{\partial h}
        \end{pmatrix}=\begin{pmatrix}
            R & 0\\
            0 & 1
        \end{pmatrix}$$
        Пусть $X=a\frac{\partial r}{\partial\alpha}+b\frac{\partial r}{\partial h}$, $Y=c\frac{\partial r}{\partial\alpha}+d\frac{\partial r}{\partial h}$, тогда $(X, \ Y)=\begin{pmatrix}
            a & b\\
        \end{pmatrix}\begin{pmatrix}
            R^2 & 0\\
            0 & 1
        \end{pmatrix}\begin{pmatrix}
            c\\
            d
        \end{pmatrix}=acR^2+bd$\\
        $(d_Pf(X), \ d_Pf(Y))=\begin{pmatrix}
            aR & b\\
        \end{pmatrix}\begin{pmatrix}
            1 & 0\\
            0 & 1
        \end{pmatrix}\begin{pmatrix}
            cR\\
            d
        \end{pmatrix}=acR^2+bd$\\
        Таким образом, получили, что дифференциал отображения сохраняет скалярное произведение, а значит, по определению данное отображение является изометрией.
    \end{proof}

        \end{pmatrix}$$

        (1) Положим $\sum\limits_lg^{lm}g_{lk}=S_k^m$, $\sum\limits_lg^{lm}g_{ls}=S_s^m$ и подставим в первую сумму $m=k$, а во вторую $m=s$.\\
        Из этого следует, что $G_{sk}(x)=g_{sk}(x)$.\\
        Остаётся проверить, что если положить $G_{sk}(x)=g_{sk}(x)$, $q_s=0$, $\lambda=1$, то получим искомую поверхность, у которой $g_{ij}$ и $b_{ij}$ --- матрицы первой и второй фундаментальных форм соответственно.
        Проверим это, $g_{ij}(x)=(r_i, \ r_j)$, $(m, \ r_j)=0$, $(m, \ m)=1$, причём $\frac{\partial r_s}{\partial x^j}=\frac{\partial^2 r}{\partial x^s\partial x^j}=\sum\limits_k\Gamma^k_{sj}r_k+b_{sj}m$ $\Longrightarrow$ $b_{js}=(m, \ r_{js})$, значит, $b_{js}$ --- матрица второй фундаментальной формы поверхности $M$.
    \end{proof}
    \begin{consequence}[Из соотношения Гаусса]
        $$\frac{\partial\Gamma^k_{js}}{\partial x^i}-\frac{\partial\Gamma_{is}^k}{\partial x^j}+\Gamma_{js}^l\cdot\Gamma_{li}^k-\Gamma_{is}^l\cdot\Gamma^k_{lj}=B_i^k\cdot b_{js}-B_j^k\cdot b_{is}$$
        Так как левая часть равенства характеризует внутреннюю геометрию поверхности, а правая --- внешнюю, то данное уравнение показывает связь внешней и внутренней геометрии.\\
        Обозначим $R_{ij, \ s}^k=B_i^k\cdot b_{js}-B_j^k\cdot b_{is}$. Этот тензор называется тензором Римана. Сделаем из него скалярную величину.
        $$R=\sum\limits_{j, \ s}g^{js}R_{kj, \ s}^k=\sum\limits_{j, \ s}g^{js}\left(\sum\limits_k\left(B^k_kb_{js}-B^k_jb_{ks}\right)\right)=\sum\limits_{k, \ j}\left(B_k^kB_j^j-B_j^kB^j_k\right)=(tr B)^2-tr(B^2)=$$
        $$=\left(\sum\limits_{j=1}^n\lambda_j\right)^2-\sum\limits_{j=1}^n\lambda_j^2=2\sum\limits_{i<j}\lambda_i\lambda_j$$
    \end{consequence}
    \begin{subtheorem}[Блистательная теорема Гаусса]
        Пусть $n=2$, тогда $k=\frac{R}{2}$, где $k$ --- гауссова кривизна.
    \end{subtheorem}
    \section{Гладкие многообразия}
    \subsection{Напоминаниме (На экзамене скорее всего доказывать не потребуется)}
    \begin{definition}
        Топологическое пространство - множество $X$ вместе с семейством $\tau$ его подмножеств, удовлетворяющее следующим условиям:\begin{enumerate}
            \item $\varnothing\in\tau$, $X\in\tau$
            \item если $\mathcal{U}\subset\tau$, то $\bigcup\mathcal{U}\in\tau$ ($\mathcal{U}$ --- семейство подмножеств $X$)
            \item $\forall U$, $V\in\tau$ $U\cap V\in\tau$
        \end{enumerate}
    \end{definition}
    \begin{definition}
        Окрестностью точки $x$ назовём любое открытое множество, содержащее $x$.
    \end{definition}
    \begin{definition}
        Пусть $X$ и $Y$ --- два топологических пространства, тогда отображение $f:X\to Y$ называется непрерывным в точке $x_0\in X$, если для любой открытой окрестности $U$ точки $x_0$ существует открытая окрестность $V$ точки $f(x_0)$ такая, что $f(U)\subset V$.
    \end{definition}
    \begin{subtheorem}
        Отображение $f$ двух топологических пространств непрерывно $\Longleftrightarrow$ прообраз любого открытого множества открыт.
    \end{subtheorem}
    \begin{proof}
        $\underline{\Longrightarrow}$ Пусть $f: X \to Y$ и пусть $A \subset Y$ открыто. Рассмотрим $f^{-1}(A)$. Докажем, что $f^{-1}(A)$ открыто в $X$. Пусть $x_0 \in f^{-1}(A)$, тогда $f(x_0) \in A$ $\Longrightarrow \exists U$ — открытая окрестность $x_0$ (по определению непрерывности): $f(U) \subset A \Longrightarrow U \subset f^{-1}(A) \Longrightarrow x_0$ — внутренняя точка $f^{-1}(A)$. В силу произвольности $x_0$ получаем, что $f^{-1}(A)$ открыто. \\
        $\underline{\Longleftarrow}$ Пусть прообраз любого открытого множества открыт. Пусть $x_0 \in X \Longrightarrow f(x_0) \in Y$. Возьмём $V \subset Y$, которое будет открыто: $f(x_0) \in V \Longrightarrow f^{-1}(V)$ — открытое множество и $x_0 \in f^{-1}(V) \Longrightarrow U := f^{-1}(V)$. Тогда $U$  — открытая окрестность $x_0$ и $f(U) = f(f^{-1}(V)) \subset V$. Получается, для любой окрестности $V$ точки $f(x_0)$ найдена окрестность $U$ точки $x_0$ такая, что $f(U) \subset V$ — это и есть определение непрерывности $f$ в точке $x_0$. Так как $x_0$ произвольна, то $f$ непрерывна на всём $X$.
    \end{proof}
    \begin{definition}
        Множество $Y\subset X$ называется компактным, если в любом его покрытии можно выделить конечное подпокрытие.
    \end{definition}
    \begin{subtheorem}
        Непрерывный образ компакта --- компакт.
    \end{subtheorem}
    \begin{proof}
        Пусть $f: X \to Y$ непрерывно. Покрываем образ открытыми множествами $\{U_{\alpha}\}$: $Im f \subseteq \bigcup\limits_{\alpha} U_{\alpha}$ — покрытие.
        $X \subset \bigcup\limits_{\alpha} f^{-1}(U_{\alpha})$ — открытое покрытие $X$ (т.к. $f$ непрерывно).
        $X \subset \bigcup\limits_{i = 1}^{n} f^{-1}(U_{i})$ — конечное подпокрытие (поскольку $X$ — компакт по условию).
        Тогда $Imf \subset \bigcup\limits_{i = 1}^n U_{i}$. Действительно, для любого $y \in f(X)$ существует $x \in X$ такой, что $f(x) = y$. Этот $x$ принадлежит некоторому $f^{-1}(U_{\alpha_i}) \Longrightarrow y \in U_{\alpha_i}$.
        %Пользуемся компактностью $X$: $\Img f \subset \bigcup_{i = 1}^{n} f(U_{i})$
    \end{proof}
    \begin{theorem} [Критерии замкнутости множества]
        Следующие условия эквивалентны:
        \begin{itemize}
            \item[(0)] $A\subset \R$ - замкнуто. 
            \item[(1)] $\partial A\subset A$,
            \item[(2)] Все точки прикосновения содержатся в $A$,
            \item[(3)] $A^{\prime}\subset A$.
        \end{itemize}
    \end{theorem}
    \begin{proof}
        Докажем по цепочке $(0)\Rightarrow (1)\Rightarrow (2)\Rightarrow (3)\Rightarrow (0)$.
        \begin{enumerate}
            \item $(0)\Rightarrow (1):A$ - замкнуто $\Rightarrow \R\setminus A$ - открыто $\Rightarrow \partial A \not\subset\R\setminus A \Rightarrow \partial A\subset A$.
            \item $(1)\Rightarrow (2):$ Все точки прикосновения являются граничными или внутренними. Поскольку $\partial A\subset A$ то все точки прикосновения содержатся в $A$.
            \item $(2)\Rightarrow (3):$ Если $x$ - предельная, то $x\in A$ или $x$ - точка прикосновения. Поскольку все точки прикосновения содержатся в $A$, то и все предельные точки содержатся в $A$.
            \item $(3)\Rightarrow (0): A^{\prime}\subset A\Rightarrow \forall x\in \R\setminus A: x\not\in A^{\prime}\Rightarrow \forall x\in\R\setminus A\ \exists\ \mathring{B}_{\epsilon}: \mathring{B}_{\epsilon}(x)\cap A=\emptyset\\
            \Rightarrow B_{\epsilon}(x)\cap A=\emptyset\ (\text{т.к}\ x\not\in A) \Rightarrow x$ - внешняя точка $A,\ B_{\epsilon}(x)\subset\R\setminus A\\
            \Rightarrow \R\setminus A$ - открыто $\Rightarrow A$ - замкнуто. 
        \end{enumerate}
    \end{proof}
    \begin{lemma}
        Пусть $K$ компакт, тогда любая бесконечная последовательность $\{x_k\}\subset K$ имеет хотя бы одну предельную точку, принадлежащую $K$.
    \end{lemma}
    \begin{proof}
        Предположим, что существует бесконечная последовательность $\{x_k\}\subset K$, которая не имеет предельных точек, принадлежащих $K$, тогда любая точка $x\in K$ имеет окрестность $U(x, \ \epsilon)$, внутри которой не будет точек из этой последовательности (по определению предельной точки, если точка $x$ --- не предельная точка последовательности, то любая её окрестность содержит конечное число точек из этой последовательности, значит, можно выбрать окрестность, не содержащую другие точки последовательности), кроме, возможно, самой точки $x$. Тогда семейство всех таких окрестностей образует покрытие множества $K$ (так как любая точка $x\in K$ имеет окрестность из данного семейства), значит, по определению компакта существует конечное подпокрытие. Так как никакая из окрестностей, семейство которых образует покрытие, не может содержать больше одной точки из последовательности, то в любом подпокрытии должны быть все окрестности с центрами в точках $x\in\{x_k\}$, но тогда в существующем конечном подпокрытии должны быть все окрестности с центрами $x\in\{x_k\}$, а значит, число точек этой последовательности, принадлежащих $K$, конечное число --- противоречие.
    \end{proof}
    \begin{subtheorem}
        Подмножество в $\R^n$ компактно тогда и только тогда, когда оно замкнуто и ограничено.
    \end{subtheorem}
    \begin{proof}
        $\underline{\Longrightarrow}$ Пусть $K\subset\R^n$ --- компакт, рассмотрим точку $x_0\in K$, тогда семейство шаров вида $U_n=O(x_0, \ k)$ (шары с центром в точке $x_0$ и радиуса $k$) покрывает всё $\R^n$ (так как это база топологии метризуемого пространства). Тогда по определению компакта существует конечное подсемейство таких шаров, покрывающих $K$, а значит, существует множество $U=\bigcup\limits_{k=1}^NU_k$ такое, что $K\subset U$, то есть $K$ ограничено.\\
        Пусть $x_0$ --- предельная точка компакта $K$, тогда по определению существует последовательность точек $\{x_k\}\subset K$ такая, что $x_k\neq x_m$ при $k\neq m$ и $x_k\to x_0$ при $k\to\infty$. Тогда по лемме $x_0\in K$, а значит, по критерию замкнутости множества $K$ замкнуто.\\
        $\underline{\Longleftarrow}$ Предположим, что $K$ замкнуто, ограничено, но не компакт. Пусть $K\subset\Pi$, где $\Pi$ некоторый куб, и пусть в некотором его открытом покрытии нельзя выделить конечное открытое подпокрытие. Рассмотрим разбиение этого куба на $2^k$ кубов, выберем из них куб, не имеющий конечного подпокрытия, и повторим операцию. Получим последовательность вложенных кубов, которая сходится к некоторой точке $x_0$, то есть $x_0$ --- предельная точка $K$, причём $x_0\in K$ по критерию замкнутости. Рассмотрим некоторую окрестность $U$ точки $x_0$, тогда она покрывает некоторый куб $\Pi_N$ из построенной последовательности, но тогда он имеет конечное покрытие --- противоречие.
    \end{proof}
    \begin{definition}
        Топологическое пространство $X$ называется \textit{связным}, если не существует двух открытых непустых непересекающихся множеств $A$ и $B$ таких, что $X = A \cup B$.
    \end{definition}
    \begin{subtheorem}
        Непрерывный образ связного пространства связен.
    \end{subtheorem}
    \begin{proof}
        $f: X \to Y$, где $X$ связно. От противного. Пусть образ несвязен. Тогда $Imf := f(X) = A \cup B$, где A, B — открытые и непустые множества, $A \cap B = \emptyset$. Рассмотрим прообразы $f^{-1}(A)$ и $f^{-1}(B)$, которые открыты в силу непрерывности отображения. Если множества не пересекаются, то и их образы не пересекаются (иначе существовала бы точка $x \in f^{-1}(A) \cap f^{-1}(B) \Longrightarrow f(x) \in A \cap B$, но $A \cap B = \emptyset$). Так как множества не пусты, то и их образы не пусты. $f^{-1}(A) \cup f^{-1}(B) = X \Longrightarrow X$ не связно — противоречие. 
    \end{proof}
    \begin{definition}
        Топологическое пространство называется \textit{линейно связным}, если любые две его точки можно соединить кривой.
    \end{definition}
    \begin{subtheorem}
        Непрерывный образ линейно связного пространства линейно связен.
    \end{subtheorem}
    \begin{proof}
    % Композиция непрерывных отображений непрерывна:
    % $$\gamma: [\alpha, \beta] \to X, \ f: X \to Y.$$
        Пусть $y_1, y_2 \in f(X)$ — произвольные точки в образе пространства $X$. Тогда $$\exists x_1, x_2 \in X: \ f(x_1) = y_1, \ f(x_2) = y_2.$$
        Так как $X$ линейно связно, то существует непрерывный путь 
        \[\gamma: [0,1] \to X,\]
        соединяющий $x_1, x_2$, то есть $\gamma(0) = x_1, \ \gamma(1) = x_2$.
        Рассмотрим композицию
        \[f \circ \gamma: [0,1] \to Y.\]
        Эта композиция непрерывна как композиция непрерывных отображений и соединяет $y_1, y_2$, потому что 
        \[(f \circ \gamma)(0) = f(\gamma(0)) = f(x_1) = y_1,\]
        \[(f \circ \gamma)(1) = f(\gamma(1)) = f(x_2) = y_2.\]
        Получаем, что для любых двух точек $y_1, y_2 \in f(X)$ мы нашли непрерывный путь в $f(X)$, их соединяющий. Следовательно, $f(X)$ линейно связно.
    \end{proof}
    \begin{definition}
        Топологическое пространство $X$ называется \textit{хаусдорфовым}, если у любых двух его различных точек существуют непересекающиеся окрестности.
    \end{definition}
    Далее рассматриваем только хаусдорфовы пространства.
    \begin{definition}
        Отображение $f:X\to Y$ двух топологических пространств называется гомеоморфизмом, если оно биективно, непрерывно и обратное к нему тоже непрерывно.
    \end{definition}
    \begin{definition}
        Пусть $M$ --- топологическое пространство, причём $M$ хаусдорфово и обладает счётной базой. $M$ называется топологическим многообразием, если у любой точки существует окрестность, гомеоморфная открытой области в $\R^n$.
    \end{definition}
    \begin{definition}
        Пусть $M$ --- $n$-мерное топологическое многообразие. Картой многообразия $M$ называется пара $(U, \ \phi)$, где $\phi:U\to\R^n$ --- гомеоморфизм.
    \end{definition}
    \begin{definition}
        Система $\{(U_{\alpha}, \ \phi_{\alpha})\}$, где $\bigcup\limits_{\alpha}U_{\alpha}\subset M$, называется атласом.
    \end{definition}
    Пусть заданы две карты $(U_{\alpha}, \ \phi_{\alpha})$ и $(U_{\beta}, \ \phi_{\beta})$, $U_{\alpha\beta}=U_{\alpha}\cap U_{\beta}\neq\varnothing$, $V_{\alpha\beta}=\phi_{\alpha}(U_{\alpha\beta})$, $V_{\beta\alpha}=\phi_{\beta}(U_{\alpha\beta})$. Рассмотрим $\phi_{\alpha\beta}:V_{\alpha\beta}\to V_{\beta\alpha}$, $\phi_{\beta\alpha}:V_{\beta\alpha}\to V_{\alpha\beta}$, $\phi_{\alpha\beta}=\phi_{\beta}\circ\phi_{\alpha}^{-1}$, $\phi_{\beta\alpha}=\phi_{\alpha}\circ\phi_{\beta}^{-1}=\phi_{\alpha\beta}^{-1}$.
    \begin{definition}
        Функции $\phi_{\alpha\beta}$ и $\phi_{\beta\alpha}$ называются функциями склейки поверхности $M$ или функциями перехода из $U_{\alpha}$ в $U_{\beta}$.
    \end{definition}
    \begin{definition}
        Пусть $(U_{\alpha}, \ \phi_{\alpha})$ --- атлас. Атлас называется гладким, если все функции склейки между его картами являются бесконечно дифференцируемыми.
    \end{definition}
    \begin{remark}
        Пусть $(U_{\alpha}, \ \phi_{\alpha})$ --- гладкий атлас и $x_{\beta}=x_{\beta}(x_{\alpha})$ --- функция склейки, тогда $det\left(\frac{\partial x_{\beta}}{\partial x_{\alpha}}\right)\neq0$ (якобиан).
    \end{remark}
    \begin{definition}
        Пусть $\{U^1_{\alpha}, \ \phi^1_{\alpha}\}$, $\{U^2_{\alpha}, \ \phi^2_{\alpha}\}$ --- два гладких атласа. Скажем, что эти два атласа эквивалентны (согласованы), если их объединение --- гладкий атлас
    \end{definition}
    \begin{definition}
        Гладкая структура на многообразии $M$ --- это класс эквивалентности гладких атласов.
    \end{definition}
    \begin{examples}\tab
        \begin{enumerate}
            \item $\R^n$
            \item $S^1\subset\R^2:x^2+y^2=1$ с функциями склейки --- стереографическими проекциями через северный и южный полюс.
            \item $\R P^2:$ $(x_0, \ x_1, \ x_2)\sim(\lambda x_0, \ \lambda x_1, \ \lambda x_2)$ для $\lambda\neq0$ и $(x_0, \ x_1, \ x_2)\neq0$.
            Тогда рассмотрим $U_1$ --- множество прямых, пересекающих плоскость $x_1=1$, $U_2$ --- множество прямых, пересекающих плоскость $x_2=1$, $U_3$ --- множество прямых, пересекающих плоскость $x_3=1$. Рассмотрим отображения $\phi_i:U_i\to\R^2$. Эти отображения являются гомеоморфизмом.\\
            Пусть $l\in U_1\cap U_2$, тогда она пересекает $P_1$ в точке $(1, \ y_2, \ y_3)$ и $P_2$ в точке $(y_1, \ 1, \ y_3)$ $\Longrightarrow$ $y_1=\frac{1}{y_2}$, $y_3=y_3$. Это отображение гладкое, значит, склейка $U_1$ и $U_2$ гладкая. Аналогично для двух других. Значит, построенный атлас является гладким, а значит, мы построили гладкую структуру.
        \end{enumerate}
    \end{examples}
